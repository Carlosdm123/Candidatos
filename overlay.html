let visible=false; let changing=false; function hideImmediately(){ const banner=document.getElementById("banner"); banner.style.opacity="0"; } hideImmediately(); channel.subscribe("banner",(message)=>{ const data=message.data; if(data.action==="show"){ const banner=document.getElementById("banner"); const text=data.text; const candidate=data.candidate; const candidateImg=document.getElementById("candidateImg"); candidateImg.src="assets/"+candidate; if(visible){ changing=true; hideBanner(); setTimeout(()=>{ document.getElementById("bannerText").innerText=text; banner.style.opacity="1"; startBanner(); visible=true; changing=false; },3500); }else{ document.getElementById("bannerText").innerText=text; banner.style.opacity="1"; startBanner(); visible=true; } } if(data.action==="hide"){ if(!visible) return; hideBanner(); setTimeout(()=>{ hideImmediately(); visible=false; },1500); } });
